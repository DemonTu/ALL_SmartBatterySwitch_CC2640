172 __STATIC_INLINE uint32_t
173 AONBatMonBatteryVoltageGet(void)
174 {
175  uint32_t ui32CurrentBattery;
176 
177  ui32CurrentBattery = HWREG(AON_BATMON_BASE + AON_BATMON_O_BAT);
178 
179  //
180  // Return the current battery voltage measurement.

205 __STATIC_INLINE bool
206 AONBatMonNewBatteryMeasureReady(void)
207 {
208  bool bStatus;
209 
210  //
211  // Check the status bit.
212  //
213  bStatus = HWREG(AON_BATMON_BASE + AON_BATMON_O_BATUPD) &
214  AON_BATMON_BATUPD_STAT ? true : false;
215 
216  //
217  // Clear status bit if set.
218  //
219  if(bStatus)
220  {
221  HWREG(AON_BATMON_BASE + AON_BATMON_O_BATUPD) = 1;
222  }
223 
224  //
225  // Return status.
226  //
227  return (bStatus);
228 }
